 94 function destroyCanvas() {
 95 	var element = document.getElementById("canvas");
 96 	element.parentNode.removeChild(element);
 97 }

 99 function clearCanvas() {
100 	// Remove the SVG element (canvas)
101 	destroyCanvas();
102 
103 	// Insert the SVG element (canvas)
104 	createCanvas();

126 function zoomed() {
127 //console.log('PAN_AND_ZOOM: ' + PAN_AND_ZOOM);
128 //console.log('zoomed. translate: ' + d3.event.translate + " scale: " + d3.event.scale);
129 	if (PAN_AND_ZOOM) {
130 		container.attr("transform", "translate(" + d3.event.translate + ")scale(" + d3.event.scale + ")");
131 		updateStatusBar();
132 		//document.getElementById('statusBar').innerHTML = 'Zoom: ' + Math.floor(d3.event.scale*100) + '%';
133 	}
134 }
135 
136 function dragstarted(d) {
137 	if (PAN_AND_ZOOM) {
138 		d3.event.sourceEvent.stopPropagation();
139 		d3.select(this).classed("dragging", true);
140 	}
141 } 
142 
143 function dragended(d) {
144 	if (PAN_AND_ZOOM) {
145 		d3.select(this).classed("dragging", false);
146 	}
147 }
