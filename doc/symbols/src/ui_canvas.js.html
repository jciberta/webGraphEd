203 /**
204  * Zooms in the layout.
205  */
206 function zoomIn() {
207 	var pz = getPanAndZoom();
208 	pz.scale += 0.5;
209 	if (pz.scale > 10) pz.scale = 10;
210 	zoom.scale(pz.scale);
211 	container.attr("transform", "translate(" + [pz.translate.x, pz.translate.y] + ")scale(" + pz.scale + ")");
212 	layout.center();
213 	updateStatusBar(); 
214 }
215 
216 function zoomOut() {
217 	var pz = getPanAndZoom();
218 	pz.scale -= 0.5;
219 	if (pz.scale < 1) pz.scale = 1;
220 	zoom.scale(pz.scale);
221 	container.attr("transform", "translate(" + [pz.translate.x, pz.translate.y] + ")scale(" + pz.scale + ")");
222 	layout.center();
223 	updateStatusBar();
224 }

233 		.attr("r", function(d) {

235 			return d.shape == 'Circle' ? ((d.visible || d.visible==undefined) ? (d.collapsed ? 8 : 5) : 0) : 0; 
236 		})

255 	node.append("rect")
256 		.attr("x", function(d) { return (d.collapsed ? -8 : -5); })

258 		.attr("width", function(d) {

263 		})
