  2 /**
  3  * Create a Reingold Tilford Tree layout
  4  * @constructor
  5  * @param {string} canvas - Canvas where the graph drawing will be lay out.
  6  * @param {GraphDrawing} graph - Graph drawing object
  7  */
  8 ReingoldTilfordTreeLayout = function(canvas, graph) {
  9 
 10  
 11   var jsonList = graph.getTreeD3JSON();
 12 
 13 console.log('jsonList: ' + jsonList);
 14 
 15 
 16   var diameter = 960;
 17 
 18   var tree = d3.layout.tree()
 19       .size([360, diameter / 2 - 120])
 20       .separation(function(a, b) { return (a.parent == b.parent ? 1 : 2) / a.depth; });
 21 
 22   var diagonal = d3.svg.diagonal.radial()
 23       .projection(function(d) { return [d.y, d.x / 180 * Math.PI]; });
 24 
 25 /*var svg = d3.select("body").append("svg")
 26     .attr("width", diameter)
 27     .attr("height", diameter - 150)
 28   .append("g")
 29     .attr("transform", "translate(" + diameter / 2 + "," + diameter / 2 + ")");
 30 */
 31 
 32   var group = canvas.append("g")
 33 //    .attr("width", diameter)
 34 //    .attr("height", diameter - 150)
 35     .attr("transform", "translate(" + diameter / 2 + "," + diameter / 2 + ")");
 36 
 37   var nodes = tree.nodes(jsonList),
 38       links = tree.links(nodes);
 39 
 40   var link = group.selectAll(".link")
 41       .data(links)
 42     .enter().append("path")
 43       .attr("class", "link")
 44       .attr("d", diagonal);
 45 
 46   var node = group.selectAll(".node")
 47       .data(nodes)
 48     .enter().append("g")
 49       .attr("class", "node")
 50       .attr("transform", function(d) { return "rotate(" + (d.x - 90) + ")translate(" + d.y + ")"; })
 51 
 52   node.append("circle")
 53       .attr("r", 4.5);
 54 
 55   node.append("text")
 56       .attr("dy", ".31em")
 57       .attr("text-anchor", function(d) { return d.x < 180 ? "start" : "end"; })
 58       .attr("transform", function(d) { return d.x < 180 ? "translate(8)" : "rotate(180)translate(-8)"; })
 59       .text(function(d) { return d.name; });
 60 
 61   //d3.select(self.frameElement).style("height", diameter - 150 + "px");
 62 
 63 }
