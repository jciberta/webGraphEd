  7 /**
  8  * Creates a node for a tree.
  9  * @class A tree is composed by nodes. TreeNode implements this behaviour with 2 properties:
 10  * <ul>
 11  * <li>node: then node itself.
 12  * <li>listChildren: list of children nodes. These are also TreeNode.
 13  * </ul>
 14  * @constructor
 15  * @param {Node} node Node of the tree.
 16  */
 17 TreeNode = function(node) {
 18     this.node = node;
 19     this.listChildren = [];
 20 };
 21 
 22 /**
 23  * Add tree node children to the tree recursively.
 24  * @param {GraphDrawing} graph The graph drawing data to populate the tree.
 25  */
 26 TreeNode.prototype.addChildren = function(graph) {
 27     var i, tn, sonId, newTreeNode;
 28 //    console.log('graph.listNodes: ' + graph.listNodes);
 29 //    console.log('graph.listEdges: ' + graph.listEdges);
 30 
 31     for (i=0;i<graph.listEdges.length;i++) {
 32 //console.log('this.node: ' + this.node);
 33         if (this.node[0] == graph.listEdges[i][0]) {
 34             // Add node
 35             sonId = graph.listEdges[i][1];
 36 //console.log('sonId: ' + sonId);
 37             newTreeNode = new TreeNode(graph.getNode(sonId));
 38             this.listChildren.push(newTreeNode);
 39 //            console.log('newTreeNode.node: ' + newTreeNode.node);
 40             newTreeNode.addChildren(graph);
 41         }
 42     }
 43 };

 45 /**
 46  * Indicates if a tree nodes has children.
 47  * @return {boolean} True if the tree nodes has children.
 48  */
 49 TreeNode.prototype.hasChildren = function() {
 50     return (this.listChildren.length>0);
 51 };

 53 /**
 54  * Convert the tree into D3 JSON notation.
 55  * @return {string} The nested JSON string.
 56  */
 57 TreeNode.prototype.toD3JSONString = function() {
 58     var i, sReturn;
 59 
 60     sReturn = '{';
 61     sReturn += '"id": "' + this.node[0]+ '", ';
 62     sReturn += '"name": "' + this.node[1]+ '"';
 63 //console.log('** this.node: ' + this.node);
 64     if (this.hasChildren()) {
 65         sReturn += ', "children": [';
 66 //console.log('** this.listChildren.length: ' + this.listChildren.length);
 67         for (i=0;i<this.listChildren.length;i++) {
 68         //    console.log('** this.listChildren[i]: ' + this.listChildren[i]);
 69             sReturn += this.listChildren[i].toD3JSONString();
 70             if (i<this.listChildren.length-1)  
 71             sReturn += ', ';
 72         }
 73         sReturn += ']';
 74     }
 75     else {
 76         sReturn += ', "size": 1';
 77     }
 78     sReturn += '}';
 79     return sReturn;
 80 };
