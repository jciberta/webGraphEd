  2 /**
  3  * Create a Collapsible Tree layout
  4  * @constructor
  5  * @param {string} canvas - Canvas where the graph drawing will be lay out.
  6  * @param {GraphDrawing} graph - Graph drawing object
  7  */
  8 CollapsibleTreeLayout = function(canvas, graph) {
  9 	var m = [20, 120, 20, 120], i = 0;
 10 
 11 	var jsonList = graph.getTreeD3JSON();
 12 	
 13 	var tree = d3.layout.tree()
 14 		.size([HEIGHT - m[1] - m[3], WIDTH - m[0] - m[2]]);
 15 
 16 	var diagonal = d3.svg.diagonal()
 17 		.projection(function(d) { return [d.y, d.x]; });
 18

 22 	
 23 console.log('CollapsibleTreeLayout');
 24   root = jsonList;
 25   root.x0 = HEIGHT / 2;
 26   root.y0 = 0;
 27 
 28   
 29 function update(source) {
 30   var duration = d3.event && d3.event.altKey ? 5000 : 500;
 31 
 32   // Compute the new tree layout.
 33   var nodes = tree.nodes(root).reverse();
 34 
 35   // Normalize for fixed-depth.
 36   nodes.forEach(function(d) { d.y = d.depth * 180; });
 37 
 38   // Update the nodes…
 39   var node = vis.selectAll("g.node")
 40       .data(nodes, function(d) { return d.id || (d.id = ++i); });
 41 
 42   // Enter any new nodes at the parent's previous position.
 43   var nodeEnter = node.enter().append("svg:g")
 44       .attr("class", "node")
 45       .attr("transform", function(d) { return "translate(" + source.y0 + "," + source.x0 + ")"; })
 46       .on("click", function(d) { toggle(d); update(d); });
 47 
 48   nodeEnter.append("svg:circle")
 49       .attr("r", 1e-6)
 50       .style("fill", function(d) { return d._children ? "lightsteelblue" : "#fff"; });
 51 
 52   nodeEnter.append("svg:text")
 53       .attr("x", function(d) { return d.children || d._children ? -10 : 10; })
 54       .attr("dy", ".35em")
 55       .attr("text-anchor", function(d) { return d.children || d._children ? "end" : "start"; })
 56       .text(function(d) { return d.name; })
 57       .style("fill-opacity", 1e-6);
 58 
 59   // Transition nodes to their new position.
 60   var nodeUpdate = node.transition()
 61       .duration(duration)
 62       .attr("transform", function(d) { return "translate(" + d.y + "," + d.x + ")"; });
 63 
 64   nodeUpdate.select("circle")
 65       .attr("r", 4.5)
 66       .style("fill", function(d) { return d._children ? "lightsteelblue" : "#fff"; });
 67 
 68   nodeUpdate.select("text")
 69       .style("fill-opacity", 1);
 70 
 71   // Transition exiting nodes to the parent's new position.
 72   var nodeExit = node.exit().transition()
 73       .duration(duration)
 74       .attr("transform", function(d) { return "translate(" + source.y + "," + source.x + ")"; })
 75       .remove();
 76 
 77   nodeExit.select("circle")
 78       .attr("r", 1e-6);
 79 
 80   nodeExit.select("text")
 81       .style("fill-opacity", 1e-6);
 82 
 83   // Update the links…
 84   var link = vis.selectAll("path.link")
 85       .data(tree.links(nodes), function(d) { return d.target.id; });
 86 
 87   // Enter any new links at the parent's previous position.
 88   link.enter().insert("svg:path", "g")
 89       .attr("class", "link")
 90       .attr("d", function(d) {
 91         var o = {x: source.x0, y: source.y0};
 92         return diagonal({source: o, target: o});
 93       })
 94     .transition()
 95       .duration(duration)
 96       .attr("d", diagonal);
 97 
 98   // Transition links to their new position.
 99   link.transition()
100       .duration(duration)
101       .attr("d", diagonal);
102 
103   // Transition exiting nodes to the parent's new position.
104   link.exit().transition()
105       .duration(duration)
106       .attr("d", function(d) {
107         var o = {x: source.x, y: source.y};
108         return diagonal({source: o, target: o});
109       })
110       .remove();
111 
112   // Stash the old positions for transition.
113   nodes.forEach(function(d) {
114     d.x0 = d.x;
115     d.y0 = d.y;
116   });
117 }

120 function toggle(d) {
121   if (d.children) {
122     d._children = d.children;
123     d.children = null;
124   } else {
125     d.children = d._children;
126     d._children = null;
127   }
128 }
