 12 function setNodeAttribute(node, attribute, value) {
 13   if (attribute == "class")
 14     node.className = value;
 15   else if (attribute == "checked")
 16     node.defaultChecked = value;
 17   else if (attribute == "for")
 18     node.htmlFor = value;
 19   else if (attribute == "style")
 20     node.style.cssText = value;
 21   else
 22     node.setAttribute(attribute, value);
 23 }

 25 function dom(name, attributes) {
 26   var node = document.createElement(name);
 27   if (attributes) {
 28     forEachIn(attributes, function(name, value) {
 29       setNodeAttribute(node, name, value);
 30     });
 31   }
 32   for (var i = 2; i < arguments.length; i++) {
 33     var child = arguments[i];
 34     if (typeof child == "string")
 35       child = document.createTextNode(child);
 36     node.appendChild(child);
 37   }
 38   return node;
 39 }
