 73 GraphDrawing.prototype.getAdjacents = function(n) {
 74   var r=[], i;
 75 
 76   for (i=0; i<this.listEdges.length; i++) {
 77     if (this.listEdges[i][0]==n) {
 78       if (r.indexOf(this.listEdges[i][1]!=-1))
 79         r.push(this.listEdges[i][1]);
 80     }
 81     else if (this.listEdges[i][1]==n) {
 82       if (r.indexOf(this.listEdges[i][0]!=-1))
 83         r.push(this.listEdges[i][0]);
 84     }
 85   }
 86   return r;
 87 }

158 /**
159  * Gets the root node when the graph drawing represents a tree, otherwise it throw an exception.
160  * @return {Node} The root node.
161  */
162 GraphDrawing.prototype.getRoot = function() {
163   if (!this.isTree) throw "Not a tree."; 
164   
165   // The source must be unique, not a target from another node.
166   var root, n1, n2, i, j;
167   var bRoot = true;
168   for (i=0; i<this.listEdges.length; i++) {
169     n1 = this.listEdges[i][0];
170     bRoot = true;
171     for (j=0; j<this.listEdges.length; j++) {
172       if (i!=j) {
173         n2 = this.listEdges[j][1];
174 //console.log('i: ' + i + ', j: ' + j + ' n1: ' + n1 + ', n2: ' + n2);
175         if (n1==n2) bRoot = false;
176       }
177     }
178     if (bRoot) {
179 //        root = n1;
180 //console.log('n1: ' + n1);
181       root = this.getNode(n1);
182       break;
183     }
184 //console.log('root: ' + root);
185   }
186   return root;
187 }

232 /**
233 * Determines if a graph drawing is connected.
234 * @return {boolean} True if the graph drawing is connected.
235 */
236 GraphDrawing.prototype.isConnected = function() {
237 	// Depth-First Search (DFS) variation (non-recursive)
238 	var l=[]; // List of visited
239 	var i=0, j;
240 
241 //console.log('GraphDrawing.isConnected:');
242 //console.log(' - listNodes: ' + this.listNodes);
243 //console.log(' - listNodes.length: ' + this.listNodes.length);
244 	if (this.listNodes.length<2) return false;
245 
246 	l.push(parseInt(this.listNodes[0][0])); // First node
247 	while (i<l.length) {
248 //console.log(' - l (list of visited): ' + l);
249 		a = this.getAdjacents(l[i]);
250 //console.log(' - a (adjacents of l): ' + a);
251 		for (j=0; j<a.length; j++) {
252 			a[j] = parseInt(a[j]);
253 			if (l.indexOf(a[j])==-1) {
254 //console.log('   - l: ' + l);
255 //console.dir(l);
256 //console.log('   - a[j]: ' + a[j]);
257 //console.dir(a[j]);
258 //console.log('   - l.indexOf(a[j]): ' + l.indexOf(a[j]));
259 				l.push(a[j]);
260 			}
261 		}
262 		i++;
263 	}
264 //console.log(' - l.length: ' + (l.length));
265 //console.log(' - Result: ' + (this.listNodes.length==l.length));
266 	return (this.listNodes.length==l.length);
267 }

269 /**
270  * Determines if a there is a cicle in a graph drawing.
271  * @return {boolean} True if the graph drawing is cyclic.
272  */
273 GraphDrawing.prototype.isCyclic = function() {
274 	// Depth-First Search (DFS) variation (non-recursive)
275 	var l=[], i=0, j, cycleDetected=false;
276 
277 	if (this.listNodes.length<2) return false;
278 
279 	l.push(this.listNodes[0][0]); // First node
280 	while (i<l.length) {
281 		a = this.getDirectedAdjacents(l[i]);
282 		for (j=0; j<a.length; j++) {
283 			if (l.indexOf(a[j])==-1)
284 				l.push(a[j]);
285 			else {
286 				cycleDetected = true;
287 				break;
288 			}
289 		}
290 		i++;
291 	}
292 	return cycleDetected;
293 }

295 /**
296  * Determines if a graph drawing is a tree.
297  * @return {boolean} True if the graph drawing is a tree.
298  */
299 GraphDrawing.prototype.isTree = function() {
300     return (this.isConnected() && !this.isCyclic());
301 }

303 /**
304  * Determines if a graph drawing is a binary tree.
305  * @return {boolean} True if the graph drawing is a binary tree.
306  */
307 GraphDrawing.prototype.isBinaryTree = function() {
308 	var i, da;
309     var bIsBinaryTree = this.isTree();
310 	
311 	if (bIsBinaryTree) {
312         // Check for each node there is no more than 2 adjacent edges
313 		for (i=0; i<this.listNodes.length;i++) {
314             da = this.getDirectedAdjacents(this.listNodes[i][0]);
315             if (da.length > 2) {
316                 bIsBinaryTree = false;
317                 break;
318             }
319         }
320 	}
321     return bIsBinaryTree;
322 }
