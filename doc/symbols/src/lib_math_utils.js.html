  4 /**
  5  * Calculates cartesian x from a polar coordinates.
  6  * @param {real} r The radial coordinate.
  7  * @param {real} t The angular coordinate.
  8  * @return {real} The cartesian x coordinate.
  9  */
 10 function cartesianX(r, t) {
 11 	return r * Math.cos((t - 90) / 180 * Math.PI);
 12 }

 18 /**
 19  * Calculates cartesian y from a polar coordinates.
 20  * @param {real} r The radial coordinate.
 21  * @param {real} t The angular coordinate.
 22  * @return {real} The cartesian y coordinate.
 23  */
 24 function cartesianY(r, t) {
 25 	return r * Math.sin((t - 90) / 180 * Math.PI);
 26 }
 27 
 28 /**
 29  * Calculates radial coordinate from a cartesian coordinates.
 30  * @param {real} x The x coordinate.
 31  * @param {real} y The y coordinate.
 32  * @return {real} The radial coordinate.
 33  */
 34 function polarR(x, y) {
 35 	return Math.sqrt(x*x + y*y);
 36 }
 37 
 38 /**
 39  * Calculates angular coordinate from a cartesian coordinates.
 40  * @param {real} x The x coordinate.
 41  * @param {real} y The y coordinate.
 42  * @return {real} The angular coordinate.
 43  */
 44 function polarT(x, y) {
 45 	return Math.atan2(y, x);
 46 }
