  4 /**

 10 Benchmark = function(files) {

 16 /**

 38 /**
 39  * Loads a file.
 40  * @param {String} filename The name of the file.
 41  */
 42 Benchmark.prototype.loadFile = function(filename) {
 43 	var timerStart, loadTime, loadFileTime, textFile = '';
 44 	var layoutHorizontalTreeTime, layoutVerticalTreeTime, layoutRadialTreeTime, layoutForceDirected;
 45 
 46 	var url = String(window.location);
 47 	var path = getBaseFromURL(url);
 48 	var filenameURL = path + '/' + filename
 49 console.log('filenameURL: ' + filenameURL);
 50 
 51 	timerStart = Date.now();
 52 
 53 	var self = this;
 54 
 55 	$('#buffer').load(filenameURL, function(response, status, xhr) {
 56 		if (status == "error") {
 57 			alert('Error ' + xhr.status + ' ' + xhr.statusText);
 58 			throw 'Benchmark Aborted.'
 59 		}
 60 		self.logDiv.innerHTML += '<h3>File ' + filename + '</h3>';
 61 		self.logDiv.innerHTML += '<ul>';
 62 		textFile = document.getElementById('buffer').innerHTML;
 63 		layout.importGML(textFile);
 64 		loadFileTime = (Date.now()-timerStart)/1000;
 65 		self.logDiv.innerHTML += '<li>Load file: ' + loadFileTime + '</li>';
 66 
 67 		// HORIZONTAL_TREE
 68 		timerStart = Date.now();
 69 		clearCanvas();
 70 		layout.layoutHorizontalTree(container);
 71 		layoutHorizontalTreeTime = (Date.now()-timerStart)/1000;
 72 		self.logDiv.innerHTML += '<li>Horizontal tree: ' + layoutHorizontalTreeTime + '</li>';
 73 console.log('HORIZONTAL_TREE: ' + layoutHorizontalTreeTime + ' s.');
 74 		
 75 		// VERTICAL_TREE
 76 		timerStart = Date.now();
 77 		clearCanvas();
 78 		layout.layoutVerticalTree(container);
 79 		layoutVerticalTreeTime = (Date.now()-timerStart)/1000;
 80 		self.logDiv.innerHTML += '<li>Vertical tree: ' + layoutHorizontalTreeTime + '</li>';
 81 console.log('VERTICAL_TREE: ' + layoutVerticalTreeTime + ' s.');
 82 
 83 		// RADIAL_TREE
 84 		timerStart = Date.now();
 85 		clearCanvas();
 86 		layout.layoutRadialTree(container);
 87 		layoutRadialTreeTime = (Date.now()-timerStart)/1000;
 88 		self.logDiv.innerHTML += '<li>Radial tree: ' + layoutHorizontalTreeTime + '</li>';
 89 console.log('RADIAL_TREE: ' + layoutRadialTreeTime + ' s.');
 90 	
 91 		// FORCE_DIRECTED
 92 		timerStart = Date.now();
 93 		clearCanvas();
 94 		layout.layoutForceDirected(container);
 95 		layoutForceDirectedTime = (Date.now()-timerStart)/1000;
 96 		self.logDiv.innerHTML += '<li>Force directed: ' + layoutForceDirectedTime + '</li>';
 97 console.log('FORCE_DIRECTED: ' + layoutForceDirectedTime + ' s.');
 98 
 99 		self.logDiv.innerHTML += '</ul>';
100 	
101 		var layoutTime = {
102 			loadFile: loadFileTime, 
103 			horizontalTree: layoutHorizontalTreeTime, 
104 			verticalTree: layoutVerticalTreeTime, 
105 			radialTree: layoutRadialTreeTime, 
106 			forceDirected: layoutForceDirectedTime
107 		};
108 
109 		self.results.push({filename: filename, time: layoutTime});
110 console.log('this.results:' + JSON.stringify(self.results));
111 		
112 		layout.clear();	
113 		if (filename == self.lastFile()) {
114 			self.menuBarDiv.style.display = 'none';
115 			self.mainDiv.style.display = 'none';
116 			self.statusBarDiv.style.display = 'none'; 
117 			document.body.style.cursor = 'default';
118 
119 			showResults();
120 		}
121 	});
122 };

124 /**
125  * Show results.
126  */
127 Benchmark.prototype.showResults = function() {
128 	var i, text = 'Benchmark results\n\n';
129 
130 	for (i=0; i<this.results.length; i++) {
131 		text += 'File ' + this.results[i].filename + '\n';
132 		text += '  - LoadFile: ' + this.results[i].time.loadFile + '\n';
133 		text += '  - HorizontalTree: ' + this.results[i].time.horizontalTree + '\n';
134 		text += '  - VerticalTree: ' + this.results[i].time.verticalTree + '\n';
135 		text += '  - RadialTree: ' + this.results[i].time.radialTree + '\n';
136 		text += '  - ForceDirected: ' + this.results[i].time.forceDirected + '\n';
137 	}
138 	alert(text);
139 }

141 /**
142  * Returns the last file of the benchmark.
143  * @return {string} The last file of the benchmark.
144  */
145 Benchmark.prototype.lastFile = function() {
146 	var i = this.files.length;
147 	return i==0 ? null : this.files[i-1];
148 }
