  4   Array.prototype.indexOf = function (obj, fromIndex) {
  5     if (fromIndex == null) {
  6       fromIndex = 0;
  7     } else if (fromIndex < 0) {
  8       fromIndex = Math.max(0, this.length + fromIndex);
  9     }
 10     for (var i = fromIndex, j = this.length; i < j; i++) {
 11       if (this[i] === obj)
 12         return i;
 13     }
 14     return -1;

 18 Array.prototype.compare = function (array) {
 19     // if the other array is a falsy value, return
 20     if (!array)
 21         return false;
 22 
 23     // compare lengths - can save a lot of time
 24     if (this.length != array.length)
 25         return false;
 26 
 27     for (var i = 0, l=this.length; i < l; i++) {
 28         // Check if we have nested arrays
 29         if (this[i] instanceof Array && array[i] instanceof Array) {
 30             // recurse into the nested arrays
 31             if (!this[i].compare(array[i]))
 32                 return false;
 33         }
 34         else if (this[i] != array[i]) {
 35             // Warning - two different object instances will never be equal: {x:20} != {x:20}
 36             return false;
 37         }
 38     }
 39     return true;
 40 }

 42 function stack() {                  
 43   var elements;
 44   
 45   this.push = function(element) {
 46     if (typeof(elements) === 'undefined') {
 47       elements = [];   
 48     }                            
 49     elements.push(element);
 50   }
 51   this.pop = function() {
 52     return elements.pop();
 53   }
 54   this.stackTop = function(element) {
 55     return elements[elements.length - 1];
 56   }
 57 }

 59 function queue() {
 60   var elements;
 61   
 62   this.enqueue = function(element) {
 63     if (typeof(elements) === 'undefined') {
 64       elements = [];   
 65     }
 66     elements.push(element);                       
 67   }
 68   this.dequeue = function() {
 69     return elements.shift();                                                
 70   }
 71   this.peek = function(){
 72     return elements[0];                  
 73   }
 74 }
