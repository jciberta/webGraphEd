443 /**
444  * Deletes a node from the graph drawing
445  * @param {int} id The identification of the node.
446  */
447 CustomLayout.prototype.deleteNode = function(id) {
448 	var i;
449 
450 	this.graph.deleteNode(id);
451 
452 	for (i=this.links.length-1; i>=0; i--) {
453 		if (this.links[i].source.id == id) {
454 			this.links.splice(i, 1);
455 		} 
456 		else if (this.links[i].target.id == id) {
457 			this.links.splice(i, 1);
458 		} 
459 	}
460 	
461 	for (i=0; i<this.nodes.length; i++) {
462 		if (this.nodes[i].id == id) {
463 			this.nodes.splice(i, 1);
464 			break;
465 		}
466 	}	
467 
468 	clearCanvas();
469 	createDragLine();
470 	this.updateLayout();
471 	updateMenu(this.graph);
472 }

512 /**
513  * Toggle a node between collapsed and uncollapsed.
514  * @param {Object} d The node to toggle.
515  */
516 CustomLayout.prototype.toggle = function(d) {
517 	// Leaves are not allowed to collapse
518 	if (!d.children) return;
519 
520 	function toggle(node, v) {
521 //console.log(' - Toggle ' + node.id);	
522 //console.dir(node);
523 		if (node.children) {
524 			var i;
525 			for (i = 0; i<node.children.length; i++) {
526 				if (node.children[i].collapsed == undefined) node.children[i].collapsed = false;
527 				if (!node.children[i].collapsed)
528 					toggle(node.children[i], v)
529 				else
530 					node.children[i].visible = v;
531 			}			
532 		}
533 		node.visible = v;
534 	}
535 	
536 //console.log('d.collapsed=' + d.collapsed);	
537 	if (d.collapsed == undefined) d.collapsed = false;
538 	d.collapsed = !d.collapsed;
539 	toggle(d, !d.collapsed);
540 	d.visible = true;
541 	
542 	this.updateCollapsedLayout();	
543 }

586 		if (d.source.visible == undefined) d.source.visible = true;
587 		if (d.target.visible == undefined) d.target.visible = true;
588 		hideLink = (!d.source.visible || !d.target.visible);
589 		return hideLink ? 0 : (d.width == undefined ? 2 : d.width); 
590 	});

596 /**
597  * Checks if there is any node collapsed.
598  * @return {Boolean} True if there is any node collapsed, otherwise, false.
599  */
600 CustomLayout.prototype.isCollapsed = function() {
601 	var i;
602 	for (i=0; i<this.nodes.length; i++) {
603 		if (this.nodes[i].collapsed != undefined && this.nodes[i].collapsed)
604 			return true;
605 	}
606 	return false;
607 }
