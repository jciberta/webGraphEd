 39 function newLayout() {
 40   // Remove the SVG element
 41   var element = document.getElementById("canvas");
 42   element.parentNode.removeChild(element);
 43 
 44   // Insert the SVG element
 45   vis = d3.select("#main")
 46     .append("svg:svg")
 47     .attr("width", WIDTH)
 48     .attr("height", HEIGHT)
 49     .attr("id", "canvas")
 50   frame = d3.select("svg")
 51     .append("svg:rect")
 52       .attr("width", "100%")
 53       .attr("height", "100%")
 54       .attr("stroke", "#000")
 55       .attr("stroke-width", 1)
 56       .attr("fill", "none");
 57 
 58   // Clear the layout
 59   layout.clear();
 60 
 61   updateMenu(false);
 62 }

 64 function importFile(filename) {
 65     var ext = getFileExtension(filename).toUpperCase();
 66     filename = filename.substr(0, filename.lastIndexOf('.')); // Trim the file extension
 67     layout.setFileName(filename); 
 68 //console.log('getFileExtension: ' + ext);
 69     if (ext == 'GML') {
 70         layout.importGML(textFile);
 71     }
 72     else if (ext == 'GRAPHML') {
 73         layout.importGraphML(textFile);
 74     }
 75     else {
 76         alert("Format not supported!"); 
 77         throw "Format not supported!";
 78     }
 79 }
