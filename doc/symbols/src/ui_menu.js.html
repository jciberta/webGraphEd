131 function importFile(filename) {
132     var ext = getFileExtension(filename).toUpperCase();
133     filename = filename.substr(0, filename.lastIndexOf('.')); // Trim the file extension
134     layout.setFileName(filename); 
135 //console.log('getFileExtension: ' + ext);
136     if (ext == 'GML') {
137         layout.importGML(textFile);
138     }
139     else if (ext == 'GRAPHML') {
140         layout.importGraphML(textFile);
141     }
142     else {
143         alert("Format not supported!"); 
144         throw "Format not supported!";
145     }
146 }

148 /**
149  * Updates the menu depending on the graph drawing type.
150  * @param {GraphDrawing} graph The graph drawing.
151  */
152 function updateMenu(graph) {
153     if ((graph==true) || (graph==false)) {
154 //        menuFileProperties.setEnabled(graph);
155         menuTree.setEnabled(graph);	
156         menuVerticalTree.setEnabled(graph);
157         menuRadialTree.setEnabled(graph);
158         menuForceDirected.setEnabled(graph);
159         menuFileExportToGML.setEnabled(graph);
160         menuFileExportToGraphML.setEnabled(graph);
161     }
162     else {
163 //        menuFileProperties.setEnabled(true);
164         isTree = graph.isTree();
165         menuTree.setEnabled(isTree);	
166         menuVerticalTree.setEnabled(isTree);	
167         menuRadialTree.setEnabled(isTree);
168         menuForceDirected.setEnabled(true);
169         menuFileExportToGML.setEnabled(true);
170         menuFileExportToGraphML.setEnabled(true);
171     }
172 }
