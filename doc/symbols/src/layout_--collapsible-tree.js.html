299     // Toggle children.
300     function toggle(d) {
301         if (d.children) {
302             d._children = d.children;
303             d.children = null;
304         } else {
305             d.children = d._children;
306             d._children = null;
307         }
308     }	
309   
310     function toggleAll(d) {
311         if (d.children) {
312             d.children.forEach(toggleAll);
313             toggle(d);
314         }
315     }
