 26 function createColorPaletteDemo(colors, width, caption) {
 27 	goog.dom.getElement('cp').appendChild(goog.dom.createDom('p', null, caption));
 28 	var cp = new goog.ui.ColorPalette(colors);
 29 	cp.setSize(width); // If we only set the columns, the rows are calculated.
 30 	cp.render(goog.dom.getElement('cp'));
 31 	goog.events.listen(cp, goog.ui.Component.EventType.ACTION, onColorEvent);
 32 	goog.events.listen(cp, EVENTS, logEvent);
 33 	return cp;
 34 }

 45 function chooseNodeProperties(d) {
 46 	// Default values
 47 	if (d.shape == undefined) d.shape = 'Circle';
 48 	if (d.color == undefined) d.color = 'White';
 49 
 50 	function listShapes(d) {
 51 		var aShapes = ['Circle', 'Square'];
 52 		var i, s = '';
 53 		
 54 		for (i=0; i<aShapes.length; i++) {
 55 			if (i % 5 == 0) s += '<tr>'; 
 56 			s += '<td><input type="radio" name="shape" value="' + aShapes[i] + '" ' + (d.shape==aShapes[i] ? 'checked="checked"' : '') + '>' + aShapes[i] + '</td>';
 57 			if ((i+1) % 5 == 0) s += '</tr>'; 
 58 		}
 59 //console.log('s: ' + s);		
 60 		return s;
 61 	}
 62 	
 63 	function listColors(d) {
 64 		var aColors = ['Red', 'White', 'Cyan', 'Silver', 'Blue', 'Gray', 'DarkBlue', 'Black', 'LightBlue', 'Orange', 'Purple', 'Brown', 'Yellow', 'Maroon', 'Lime', 'Green', 'Magenta', 'Olive'];
 65 		var i, s = '';
 66 		
 67 		for (i=0; i<aColors.length; i++) {
 68 			if (i % 5 == 0) s += '<tr>'; 
 69 			s += '<td><input type="radio" name="color" value="' + aColors[i] + '" ' + (d.color==aColors[i] ? 'checked="checked"' : '') + '>' + aColors[i] + '</td>';
 70 			if ((i+1) % 5 == 0) s += '</tr>'; 
 71 		}
 72 //console.log('s: ' + s);		
 73 		return s;
 74 	}
 75 	
 76 	var txt = '<h1>Node Properties</h1>'+
 77 		'Select shape:<br>' +
 78 		'<table>' + listShapes(d) + '</table>' +
 79 		'<br>Select color:<br>' +
 80 		'<table>' + listColors(d) + '</table>';
 81 
 82 	$.prompt(txt, { 
 83 		buttons: {Change:true, Cancel:false},
 84 		submit: function(e, v, m, f) {
 85 			// This is simple pre submit validation, the submit function
 86 			// return true to proceed to the callback, or false to take 
 87 			// no further action, the prompt will stay open.
 88 			var flag = true;
 89 			// ...
 90 			return flag;
 91 		},
 92 		close: function(e, v, m, f) {
 93 			if (v) {
 94 				var n;
 95 				$.each(f, function(i, obj) {
 96 					n = layout.graph.getNode(d.id);
 97 					switch (i) {
 98 						case 'shape': 
 99 							n[2] = obj; 
100 							d.shape = obj; 
101 							layout.changeNodeShape(d); 
102 							break;
103 						case 'color': 
104 							n[3] = obj; 
105 							d.color = obj; 
106 							layout.changeNodeColor(d); 
107 							break;
108 					}
109 				});	
110 //				layout.update();
111 			}
112 			else {}
113 		}
114 	});
115 }
