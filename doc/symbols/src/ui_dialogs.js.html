 26 function createColorPaletteDemo(colors, width, caption) {
 27 	goog.dom.getElement('cp').appendChild(goog.dom.createDom('p', null, caption));
 28 	var cp = new goog.ui.ColorPalette(colors);
 29 	cp.setSize(width); // If we only set the columns, the rows are calculated.
 30 	cp.render(goog.dom.getElement('cp'));
 31 	goog.events.listen(cp, goog.ui.Component.EventType.ACTION, onColorEvent);
 32 	goog.events.listen(cp, EVENTS, logEvent);
 33 	return cp;
 34 }
