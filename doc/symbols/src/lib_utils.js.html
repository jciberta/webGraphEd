  8 function installedPlugins() {
  9   var out = "";
 10   out = out + "<table cellspacing='4' cellpadding='4'>";
 11   out = out + "<tr><th bgcolor='#CCCCCC'>Name<\/th><th bgcolor='#CCCCCC'>File<\/th><th bgcolor='#CCCCCC'>Description<\/th><\/tr>";
 12   for (i=0; i<navigator.plugins.length; i++) {
 13 	  thisPlugin = navigator.plugins[i]
 14 	  out = out + "<tr valign='top'><td bgcolor='#CCCCCC'>" + thisPlugin.name;
 15 	  out = out + "<\/td><td bgcolor='#CCCCCC'>" + thisPlugin.filename;
 16 	  out = out + "<\/td><td bgcolor='#CCCCCC'>" + thisPlugin.description + "<\/td><\/tr>";
 17   }
 18   out = out + "<\/table>";
 19 	return out;
