 83         normX = deltaX / dist,
 84         normY = deltaY / dist,
 85         sourcePadding = d.left ? 7 : 2,
 86         targetPadding = d.right ? 7 : 2,
 87         sourceX = d.source.x + (sourcePadding * normX),
 88         sourceY = d.source.y + (sourcePadding * normY),
 89         targetX = d.target.x - (targetPadding * normX),
 90         targetY = d.target.y - (targetPadding * normY);

302 VerticalTreeLayout.prototype.toggle = function(d) {
303 	if (d.children) {
304 		d._children = d.children;
305 		d.children = null;
306 	} else {
307 		d.children = d._children;
308 		d._children = null;
309 	}
