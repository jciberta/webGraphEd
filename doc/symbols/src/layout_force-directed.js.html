148 ForceDirectedLayout.prototype.createForce = function() {
149 	this.force = d3.layout.force().size([WIDTH, HEIGHT])
150 		.nodes(this.nodes)
151 		.links(this.links)
152 		.gravity(1).linkDistance(50)
153 		.charge(-3000)
154 		.linkStrength(function(x) {
155 			return x.weight * 10
156 		})
157 		.on('end', function() {
158 			// Message when the force is balanced 
159 			// (the "end" event is dispatched when the simulations internal alpha cooling parameter reaches zero)
160 			// https://github.com/mbostock/d3/wiki/Force-Layout#on
161 			statusBarMessage = 'Force directed layout stopped in ' + (Date.now()-timerStart)/1000 + ' s.';
162 			updateStatusBar();
163 		});
164 	//this.force.start();

483  * @param {int} id The id of the node.
484  * @return {Object} The node or null.
485  */
486 ForceDirectedLayout.prototype.getNode = function(id) {
487     var i;
488     for (i=0; i<this.nodes.length; i++) { 
489         if (this.nodes[i].id==id)
490 			return this.nodes[i];
491     } 
492     return null;   
493 }
