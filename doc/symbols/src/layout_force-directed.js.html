  2 /**
  3  * Create a Force Directed layout
  4  * @constructor
  5  * @param {string} canvas - Canvas where the graph drawing will be lay out.
  6  * @param {GraphDrawing} graph - Graph drawing object
  7  */
  8 ForceDirectedLayout = function(canvas, graph) {
  9 
 10 
 11   var adjacencyList = graph.getAdjacencyList();
 12 console.log('adjacencyList: ' + adjacencyList);
 13 
 14   var nodes = d3.values(adjacencyList);
 15   var links = d3.merge(nodes.map(function(source) {
 16         return source.map(function(target) {
 17           return {source: source, target: adjacencyList[target]};
 18         });
 19       }));
 20 
 21 console.log('canvas: ', canvas);
 22 
 23 /*var w = 960,
 24     h = 500;
 25 
 26 var vis = d3.select("body").append("svg:svg")
 27     .attr("width", w)
 28     .attr("height", h);*/
 29 
 30 
 31   var force = d3.layout.force()
 32       .nodes(nodes)
 33       .links(links)
 34       .size([w, h])
 35       .start();
 36 
 37   var link = canvas.selectAll("line.link")
 38       .data(links)
 39     .enter().append("svg:line");
 40 
 41   var node = canvas.selectAll("circle.node")
 42       .data(nodes)
 43     .enter().append("svg:circle")
 44       .attr("r", 5)
 45       .call(force.drag);
 46 
 47   force.on("tick", function() {
 48     link.attr("x1", function(d) { return d.source.x; })
 49         .attr("y1", function(d) { return d.source.y; })
 50         .attr("x2", function(d) { return d.target.x; })
 51         .attr("y2", function(d) { return d.target.y; });
 52 
 53     node.attr("cx", function(d) { return d.x; })
 54         .attr("cy", function(d) { return d.y; });
 55   });
 56 }
