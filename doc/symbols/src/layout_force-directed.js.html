211 			} 
212 			else {
213 				var b = this.childNodes[1].getBBox();
214 				var diffX = d.x - d.node.x;
215 				var diffY = d.y - d.node.y;
216 				var dist = Math.sqrt(diffX * diffX + diffY * diffY);
217 				var shiftX = b.width * (diffX - dist) / (dist * 2);
218 				shiftX = Math.max(-b.width, Math.min(0, shiftX));
219 				var shiftY = 5;
220 				this.childNodes[1].setAttribute("transform", "translate(" + shiftX + "," + shiftY + ")");
221 			}
