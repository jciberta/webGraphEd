107 ForceDirectedLayout.prototype.createForce = function() {
108 	this.force = d3.layout.force().size([WIDTH, HEIGHT])
109 		.nodes(this.nodes)
110 		.links(this.links)
111 		.gravity(1).linkDistance(50).charge(-3000).linkStrength(function(x) {
112 			return x.weight * 10
113 		});

376 /**
377  * Gets the node with a specific id.
378  * @param {int} id The id of the node.
379  * @return {Object} The node or null.
380  */
381 ForceDirectedLayout.prototype.getNode = function(id) {
382     var i;
383     for (i=0; i<this.nodes.length; i++) { 
384         if (this.nodes[i].id==id)
385 			return this.nodes[i];
386     } 
387     return null;   
388 }
