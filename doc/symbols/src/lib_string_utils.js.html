  3 /**
  4  * Gives format to s string like printf/string.format
  5  * http://stackoverflow.com/questions/610406/javascript-equivalent-to-printf-string-format
  6  * Example: "{0} is dead, but {1} is alive! {0} {2}".format("ASP", "ASP.NET")
  7  * Output: ASP is dead, but ASP.NET is alive! ASP {2}
  8  */
  9 String.prototype.format = function() {
 10     var formatted = this;
 11     for (var i = 0; i < arguments.length; i++) {
 12         var regexp = new RegExp('\\{'+i+'\\}', 'gi');
 13         formatted = formatted.replace(regexp, arguments[i]);
 14     }
 15     return formatted;
 16 };

 26 function getStringBetween(str, left, right) {
 27   var i, s = '';
 28 
 29   i = str.indexOf(left);
 30   if (i>0) {
 31     s = str.substring(i+left.length, str.length);
 32     i = s.indexOf(right);
 33     if (i>0) {
 34       s = s.substring(0, i);
 35     }
 36   }
 37 
 38   return s;
 39 }

 41 /**
 42  * Gets the next word from position i or from a key word.
 43  * @return {string} The next word.
 44  */
 45 function nextWord(str, i) {
 46 	var i, s = '';
 47 
 48 	if (typeof i == 'string') {
 49 		i = str.indexOf(i);
 50 		if (i == -1)
 51 			return s;
 52 	}
 53   
 54 	// Begin of the next word
 55 	i = str.indexOf(' ', i);
 56 	s = str.substring(i, str.length).trim() + ' '; 
 57 	// End of the next word
 58 	i = s.indexOf(' ');
 59 	s = s.substring(0, i).trim();  
 60 	return s;
 61 }

 63 /**
 64  * Gets the next quoted word from position i or from a key word.
 65  * @return {string} The next quoted word.
 66  */
 67 function nextQuotedWord(str, i) {
 68 	var i, s = '';
 69 
 70 	if (typeof i == 'string') {
 71 		i = str.indexOf(i);
 72 		if (i == -1)
 73 			return s;
 74 	}
 75 
 76 	// Begin of the next word
 77 	i = str.indexOf('"', i);
 78 	if (i>=0) {
 79 		s = str.substring(i+1, str.length).trim();  
 80 		// End of the next word
 81 		i = s.indexOf('"');
 82 		s = s.substring(0, i).trim();  
 83 	}
 84 	return s;
 85 }
