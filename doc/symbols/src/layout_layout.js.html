171 /**
172  * Gets the margins of the layout.
173  */
174 Layout.prototype.getLayoutMargins = function() {
175 	var xmin=0, ymin=0, xmax=0, ymax=0;
176 	var i, n, l;
177 	
178 	if (this.layout.nodes.length>0) {
179 		n = this.layout.nodes[0];
180 		xmin = n.x;
181 		xmax = n.x;
182 		ymin = n.y;
183 		ymax = n.y;
184 	}
185 	for (i=0; i<this.layout.nodes.length; i++) {
186 		n = this.layout.nodes[i];
187 		if (n.x < xmin) xmin = n.x;
188 		if (n.x > xmax) xmax = n.x;
189 		if (n.y < ymin) ymin = n.y;
190 		if (n.y > ymax) ymax = n.y;
191 	}
192 	return {xMin: xmin, xMax: xmax, yMin: ymin, yMax: ymax};
193 }

205 	updateStatusBar();
206 }

254 /**
255  * Fits the layout.
256  */
257 Layout.prototype.fit = function() {
258 	var k = 4; // k times the margin for fitting
259 	var box = this.getLayoutMargins();
260 console.log('Box:' + JSON.stringify(this.getLayoutMargins()));	
261 	
262 	var deltaX = box.xMax - box.xMin - (k*margin.left + k*margin.right);
263 	var deltaY = box.yMax - box.yMin - (k*margin.top + k*margin.bottom); 
264 	
265 	var factorX = WIDTH / deltaX;
266 	var factorY = HEIGHT / deltaY;
267 	
268 	for (i=0; i<this.layout.nodes.length; i++) {
269 		n = this.layout.nodes[i];
270 		n.x = (n.x - box.xMin) * factorX - WIDTH/2 + k*margin.left;
271 		n.y = (n.y - box.yMin) * factorY - HEIGHT/2 + k*margin.top;
272 	}
273 	
274 	
275 //console.dir(this.layout.nodes);	
276 //console.dir(this.layout.links);	
277 	
278 //	this.layout.updateLayout();
279 	var nodes = this.layout.nodes;
280 	var links = this.layout.links;
281 	clearCanvas();
282 //console.dir(nodes);	
283 //console.dir(links);	
284 
285 console.log('Box:' + JSON.stringify(this.getLayoutMargins()));	
286 console_listNodes(this.layout.nodes);
287 
288 	this.layout = new CustomLayout(canvas, this.graph, nodes, links);
289 	this.setOriginWithNoZoom();
290 	this.center();
291 	
292 };
