  2 /**
  3  * Create a Layout
  4  * @constructor
  5  */
  6 Layout = function() {
  7   this.graph = new GraphDrawing();
  8   this.file = "";
  9   this.fileName = "";
 10   this.layout = null;
 11 }
 12 
 13 /**
 14  * Clear the layout.
 15  */
 16 Layout.prototype.clear = function() {
 17   delete this.graph;
 18   this.graph = new GraphDrawing();
 19   delete this.layout;
 20   this.layout = null;
 21 };
