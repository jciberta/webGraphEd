187 /**
188  * Gets the margins of the layout.
189  */
190 Layout.prototype.getLayoutMargins = function() {
191 	var xmin=0, ymin=0, xmax=0, ymax=0;
192 	var i, n, l;
193 	
194 	if (this.layout.nodes.length>0) {
195 		n = this.layout.nodes[0];
196 		xmin = n.x;
197 		xmax = n.x;
198 		ymin = n.y;
199 		ymax = n.y;
200 	}
201 	for (i=0; i<this.layout.nodes.length; i++) {
202 		n = this.layout.nodes[i];
203 		if (n.x < xmin) xmin = n.x;
204 		if (n.x > xmax) xmax = n.x;
205 		if (n.y < ymin) ymin = n.y;
206 		if (n.y > ymax) ymax = n.y;
207 	}
208 	return {xMin: xmin, xMax: xmax, yMin: ymin, yMax: ymax};
209 }

211 /**
212  * Sets the origin to 0,0.

214 Layout.prototype.setOrigin = function() {
215 	var pz = getPanAndZoom();
216 	var coord = [-WIDTH/2*(pz.scale-1), -HEIGHT/2*(pz.scale-1)];
217 	zoom.translate(coord);

219 //console.log('setOrigin.scale: ' + pz.scale);	
220 	container.attr('transform', 'translate(' + coord + ')scale(' + pz.scale + ')');
221 	updateStatusBar();
222 }

270 /**
271  * Fits the layout.
272  */
273 Layout.prototype.fit = function() {
274 	var k = 4; // k times the margin for fitting
275 	var box = this.getLayoutMargins();
276 //console.log('Box:' + JSON.stringify(this.getLayoutMargins()));	
277 	
278 	var deltaX = box.xMax - box.xMin - (k*margin.left + k*margin.right);
279 	var deltaY = box.yMax - box.yMin - (k*margin.top + k*margin.bottom); 
280 	
281 	var factorX = WIDTH / deltaX;
282 	var factorY = HEIGHT / deltaY;
283 	
284 	for (i=0; i<this.layout.nodes.length; i++) {
285 		n = this.layout.nodes[i];
286 		n.x = (n.x - box.xMin) * factorX - WIDTH/2 + k*margin.left;
287 		n.y = (n.y - box.yMin) * factorY - HEIGHT/2 + k*margin.top;
288 	}
289 	
290 	
291 //console.dir(this.layout.nodes);	
292 //console.dir(this.layout.links);	
293 	
294 //	this.layout.updateLayout();
295 	var nodes = this.layout.nodes;
296 	var links = this.layout.links;
297 	clearCanvas();
298 //console.dir(nodes);	
299 //console.dir(links);	
300 
301 //console.log('Box:' + JSON.stringify(this.getLayoutMargins()));	
302 //console_listNodes(this.layout.nodes);
303 
304 	this.layout = new CustomLayout(canvas, this.graph, nodes, links);
305 	this.setOriginWithNoZoom();
306 	this.center();
307 	
308 };

504 /**
505  * Deletes a link from the layout
506  * @param {int} source The source of the link.
507  * @param {int} target The target of the link.
508  */
509 Layout.prototype.deleteLink = function(source, target) {
510 	var i;
511 
512 	this.graph.deleteLink(source, target);
513 	for (i=0; i<this.layout.links.length; i++) {
514 		if (this.layout.links[i].source.id == source && this.layout.links[i].target.id == target) {		
515 			this.layout.links.splice(i, 1);
516 			break;
517 		}
518 	}	
519 
520 	clearCanvas();
521 	createDragLine();
522 	this.layout.updateLayout();
523 	updateMenu(this.graph);
524 }
