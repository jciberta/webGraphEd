 36 RadialTreeLayout.prototype.transformToCartesianCoordinates = function(obj, x, y) {
 37 	var i, r = obj.y, t = obj.x;
 38 	obj.x = cartesianX(r, t);
 39 	obj.y = cartesianY(r, t);
 40 //console.log('r: ' + r + ', t: ' + t + ' -> x: ' + obj.x + ', y: ' + obj.y);		
 41 	for (i=0; i<this.links.length;i++) {
 42 		if (this.links[i].source.id == obj.id) {
 43 			this.links[i].source.x = obj.x;
 44 			this.links[i].source.y = obj.y;
 45 		}
 46 		else if (this.links[i].target.id == obj.id) {
 47 			this.links[i].target.x = obj.x;
 48 			this.links[i].target.y = obj.y;
 49 		}
 50 	}
 51 		if (obj.hasOwnProperty('children')) {
 52 		for (i=0; i<obj.children.length;i++) {
 53 			this.transformToCartesianCoordinates(obj.children[i], obj.x, obj.y);
 54 		}
 55 	}
 56 }
