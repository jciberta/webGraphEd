  2 /**
  3  * Create a Radial Tree based on the Reingold Tilford Tree layout
  4  * @constructor
  5  * @param {string} canvas - Canvas where the graph drawing will be lay out.
  6  * @param {GraphDrawing} graph - Graph drawing object
  7  */
  8 RadialTreeLayout = function(canvas, graph) {
  9 	this.className = 'RadialTreeLayout';
 10 	
 11 //console.log('RadialTreeLayout');
 12     var jsonList = graph.getTreeD3JSON();
 13 //console.log('jsonList: ' + JSON.stringify(jsonList));
 14 
 15     var diameter = HEIGHT;
 16 
 17     var tree = d3.layout.tree()
 18         .size([360, diameter / 2 - 120])
 19         .separation(function(a, b) { 
 20 //console.log('a.depth: ' + a.depth);
 21 //			return (a.parent == b.parent ? 1 : 2) / a.depth; 
 22 			return (a.parent == b.parent ? 1 : 2) / (a.depth == 0 ? 1 : a.depth ); 
 23 		});
 24 	
 25     this.nodes = tree.nodes(jsonList);
 26     this.links = tree.links(this.nodes);
 27 	
 28 //console.log('nodes:');
 29 //console.dir(this.nodes);
 30 //console.log('links:');
 31 //console.dir(this.links);
 32 
 33 	// Transform polar coordinates to cartesian coordinates in the layout
 34 	this.transformToCartesianCoordinates(this.nodes[0], 0, 0);
 35 
 36 //console.log('nodes:');
 37 //console.dir(this.nodes);
 38 //console.log('links:');
 39 //console.dir(this.links);
 40 }

 42 RadialTreeLayout.prototype.transformToCartesianCoordinates = function(obj, x, y) {
 43 	var i, r = obj.y, t = obj.x;
 44 	obj.x = cartesianX(r, t);
 45 	obj.y = cartesianY(r, t);
 46 //console.log('r: ' + r + ', t: ' + t + ' -> x: ' + obj.x + ', y: ' + obj.y);		
 47 	for (i=0; i<this.links.length;i++) {
 48 		if (this.links[i].source.id == obj.id) {
 49 			this.links[i].source.x = obj.x;
 50 			this.links[i].source.y = obj.y;
 51 		}
 52 		else if (this.links[i].target.id == obj.id) {
 53 			this.links[i].target.x = obj.x;
 54 			this.links[i].target.y = obj.y;
 55 		}
 56 	}
 57 		if (obj.hasOwnProperty('children')) {
 58 		for (i=0; i<obj.children.length;i++) {
 59 			this.transformToCartesianCoordinates(obj.children[i], obj.x, obj.y);
 60 		}
 61 	}
 62 }
