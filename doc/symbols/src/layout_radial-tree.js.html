 73 			this.links[i].source.y = obj.y;
 74 		}
 75 		else if (this.links[i].target.id == obj.id) {
 76 			this.links[i].target.x = obj.x;

125 console.log('computeTransitionPath - d:');

127 		var deltaX = d.target.x - d.source.x,
128         deltaY = d.target.y - d.source.y,
129         dist = Math.sqrt(deltaX * deltaX + deltaY * deltaY),
130         normX = deltaX / dist,
131         normY = deltaY / dist,
132         sourcePadding = d.left ? 7 : 2,
133         targetPadding = d.right ? 7 : 2,
134         sourceX = d.source.x + (sourcePadding * normX),
135         sourceY = d.source.y + (sourcePadding * normY),
136         targetX = d.target.x - (targetPadding * normX),
