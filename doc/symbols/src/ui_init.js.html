 10 function loadGraphDrawing() {
 11 	// Load file into layout object
 12 	switch (format) {
 13 		case 'GML':
 14 			layout.importGML(graph);
 15 			break;
 16 		case 'GRAPHML':
 17 			layout.importGraphML(graph);
 18 			break;
 19 		default:
 20 			alert("Format not supported!"); 
 21 			throw "Format not supported!";
 22 			break;
 23 	}
 24    
 25     // Layout the graph drawing
 26     switch (query.layout) {
 27         case 't':
 28         case 'ht':
 29             layout.layoutHorizontalTree(canvas);
 30             break;
 31         case 'vt':
 32             layout.layoutVerticalTree(canvas);
 33             break;
 34         case 'rt':
 35             layout.layoutRadialTree(container);
 36             break;
 37         case 'fd':
 38             layout.layoutForceDirected(canvas);
 39             break;
 40         default:
 41             alert("Layout not supported!"); 
 42             throw "Layout not supported!";
 43             break;
 44     }
 45 	
 46 	if (EMBEDDED && query.layout!='fd') { 
 47 		layout.fit(); 
 48 	}
 49     updateMenu(layout.graph);
 50 }
